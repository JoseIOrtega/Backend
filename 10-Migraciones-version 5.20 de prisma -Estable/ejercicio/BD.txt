model Autor {
  id     Int     @id @default(autoincrement())
  nombre String
  libros Libro[]
}

model Libro {
  id        Int        @id @default(autoincrement())
  titulo    String
  anio      Int
  autor     Autor      @relation(fields: [autorId], references: [id])
  autorId   Int
  prestamos Prestamo[]
}

model Prestamo {
  id        Int      @id @default(autoincrement())
  fecha     DateTime @default(now())
  libro     Libro    @relation(fields: [libroId], references: [id])
  libroId   Int
  usuario   String   // nombre de la persona que pidi√≥ el libro
}
